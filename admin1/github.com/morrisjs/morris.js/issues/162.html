t3chn0r commented Dec 16, 2013

Something like this perhaps?

yLabelFormat: function (y) {
    var parts = y.toString().split(".");
    parts[0] = parts[0].replace(/\B(?=(\d{3})+(?!\d))/g, ".");
    return parts.join(",");
}
